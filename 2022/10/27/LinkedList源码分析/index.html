    void linkBefore(E e, Node succ) {
    // assert succ != null;
    // 首先记录要插入到索引位置的节点的前驱
    final Node pred = succ.prev;
    // 构造方法 前驱：原节点前驱 e 需要插入节点 后继： 索引位置节点succ
    final Node newNode = new Node<>(pred, e, succ);
    // 将索引位置节点的前驱指向新节点
    succ.prev = newNode;
    if (pred == null)
    first = newNode;
    else
    pred.next = newNode;
    size++;
    modCount++;
    }
